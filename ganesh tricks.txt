    function onlyChars(e) {
        var keyVal = 0;

        if (!e.which)

            keyVal = e.keyCode;

        else

            keyVal = e.which; if (keyVal !== 27) {

                if ((keyVal >= 65) && (keyVal <= 90) || (keyVal >= 97) && (keyVal <= 122) || keyVal === 8 || keyVal === 9 || keyVal === 32 || keyVal === 38 || keyVal === 45) {

                    return true;

                } else {

                    keyVal = 0;

                    return false;

                }

            }
    }

function isEmail(email) {

CONTROLLER;

 public ActionResult DDl()
        {
            string con = ConfigurationManager.ConnectionStrings["DBCS"].ConnectionString;
            SqlConnection sqlcon = new SqlConnection(con);
            string sqlquery = "select *from [dbo].[Master_Country]";
            SqlCommand sqlcmd = new SqlCommand(sqlquery, sqlcon);
            sqlcon.Open();
            SqlDataAdapter sqlda = new SqlDataAdapter(sqlcmd);
            DataSet ds = new DataSet();
            sqlda.Fill(ds);
            ViewBag.countryname = ds.Tables[0];

            List<SelectListItem> getcountryname = new List<SelectListItem>();

            foreach (System.Data.DataRow dr in ViewBag.countryname.Rows)
            {
                getcountryname.Add(new SelectListItem { Text = dr["name"].ToString(), Value = dr["name"].ToString() });

            }
            ViewBag.countrys = getcountryname;
            sqlcon.Close();
            return View();
        }
